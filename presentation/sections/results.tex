\begin{frame}{Key Findings \& Analysis}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \begin{block}{Action Masking Benefits}
        \begin{itemize}
          \item \pro{100\% feasibility} of every accepted placement
          \item Improved sample efficiency --- no wasted gradients on infeasible actions
          \item Simplified reward design: no per-constraint penalty terms needed
        \end{itemize}
      \end{block}
      \vspace{4pt}
      \begin{block}{GNN vs.\ MLP Policies}
        \begin{itemize}
          \item GNN captures \textbf{centrality, neighborhood load}, and \textbf{security posture clusters}
          \item Attention weights reveal which nodes the policy considers most important
          \item Structural awareness unavailable to flat feature vectors
        \end{itemize}
      \end{block}
    \column{0.47\textwidth}
      \begin{block}{PPO's Long-Horizon Advantage}
        \begin{itemize}
          \item May accept \emph{slightly riskier} early placements to \textbf{preserve low-risk nodes} for future strict requests
          \item Greedy Best-Fit cannot exhibit this strategy
        \end{itemize}
      \end{block}
      \vspace{4pt}
      \begin{block}{Windowed AR Modifier}
        \begin{itemize}
          \item Amplifies acceptance reward after high-AR phases
          \item Amplifies rejection penalty after low-AR phases
          \item Provides \kw{curriculum-style} training signal
          \item Stabilizes training convergence
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}
