\begin{frame}{MDP Formulation}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \begin{block}{$\mathcal{M} = \langle \mathcal{S}, \mathcal{A}, \mathcal{P}, \mathcal{R}, \gamma \rangle$}
        \footnotesize
        \begin{itemize}
          \item \textbf{State} $\mathbf{s}_t$: node feature matrix $\mathbf{F}\in\mathbb{R}^{25\times20}$, global context $\mathbf{g}\in\mathbb{R}^{15}$, placement mask, node mask
          \item \textbf{Action}: select substrate node $n_k$ for VNF $v_k$
          \item \textbf{Episode}: $T = 300$ SFC requests; $K_r = 3$ VNFs/chain $\Rightarrow$ up to 900 steps
          \item \textbf{Discount}: $\gamma = 0.99$
        \end{itemize}
      \end{block}
      \vspace{4pt}
      \begin{block}{Objective}
        \[J(\pi) = \mathbb{E}_{\pi}\!\left[\sum_{t=0}^{T} \gamma^t r_t\right]\]
      \end{block}
    \column{0.47\textwidth}
      \begin{block}{Reward Structure}
        \footnotesize
        \textbf{Intermediate} (per VNF $k < K_r$):
        \[r_t^{\text{step}} = 0.1 - 0.3\,\frac{L(n_{k-1},n_k)}{\ell_r^{\max}}\]
        \textbf{Terminal acceptance:}
        \[r_T^{\text{acc}} = \bigl(1 - \lambda\rho_r - \mu\widehat{C}_r\bigr)\bigl(1+\omega\overline{\text{AR}}_w\bigr)\]
        \textbf{Terminal rejection:}
        \[r_T^{\text{rej}} = -1 \cdot \bigl(1+\omega(1-\overline{\text{AR}}_w)\bigr)\]
        \vspace{2pt}
        $\lambda=4$, $\mu=0.05$, $\omega=0.5$, $w=10$
      \end{block}
  \end{columns}
\end{frame}
