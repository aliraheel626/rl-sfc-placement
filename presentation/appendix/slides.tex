\begin{frame}[noframenumbering]{Appendix: Node Feature Details}
  \centering\footnotesize
  \begin{tabular}{clp{6cm}}
    \toprule
    \textbf{Index} & \textbf{Group} & \textbf{Description}\\
    \midrule
    0--3  & Base resources      & Residual RAM, CPU, Storage (fraction of initial); Effective security $s_n^{\text{eff}}/s_{\max}$\\
    4     & Connectivity        & Avg.\ residual bandwidth to neighbors / $b_{\max}$\\
    5     & Distance            & Shortest-path latency from previous placement node / $\ell_{\max}$\\

    9     & VNF global share    & Fraction of all VNF instances hosted here\\
    10    & SFC tenancy ratio   & $\mathrm{SFC}(n)/\max(\mathrm{VNF}(n),1)$: blast-radius proxy\\
    11--13& VNF fit ratios      & $d_v^r / \tilde{r}_n^r$ for RAM, CPU, Storage\\
    14    & Incident pressure   & $p_n$\\
    15    & Load norm           & $\mathrm{VNF}(n)/Q_{75}^{\mathrm{VNF}}$\\
    16    & Base incident prob. & $q_n$ (Eq.\ in System Model)\\
    17    & Expected incidents  & $q_n \cdot T_s / T_{\text{cap}}$\\
    18    & Node risk score     & $= q_n$ (explicit risk slot)\\
    19    & Expected lost rev.  & $1 - 1/(1 + P_n^{\text{hit}}\cdot\mathrm{SFC}(n)\cdot R_{\text{ttl}}\cdot D_{\text{inc}})$\\
    \bottomrule
  \end{tabular}
\end{frame}
